Server/Render objects

# Define the server logic
server <- function(input, output, session) {
  

    ###Add Map plot
  output$plot <- renderPlot({
    pres_dat %>% 
        filter(year==input$year) %>% 
      ggplot() +
  geom_polygon(aes(x = long, y = lat, fill = factor(party), group = group), color="white", alpha=0.75) + 
  coord_fixed(1.3) +
  theme_void()+
  scale_fill_manual(values=c("blue", "red"))+
  theme(legend.position = "false")+
      ggtitle(paste0("Presidential Election ", input$year))+
  theme(plot.title = element_text(size = 28, face = "bold", hjust=0.5))
    
  }, height=600)
  
  ###
  
    #Add table
    output$table <- renderTable({
    table_dat<-pres_dat %>% 
      filter(year==input$year) %>% 
            select(party, year, candidate, state, percent) %>% 
            group_by(state) %>% 
            #mutate(state_winner=ifelse(percent > 50, 1, 0)) %>% 
            ungroup() %>% 
            unique()
    
    table_dat
  }, width = "90%", height="90%")
  
   ##Action button function
   observeEvent(input$next_seg,{
    x <- as.numeric(input$year)
    #reactive expression
    n <- x+4
    updateTextInput(session, "year", value = paste(n))

    })

      ##Action button function 2
   observeEvent(input$prev,{
    x <- as.numeric(input$year)
    #reactive expression
    p <- x-4
    updateTextInput(session, "year", value = paste(p))

    })
     
  ###

  
###

   ###
    
}
